CREATE TABLE PRODUCT
(
	PRODUCTID INT PRIMARY KEY auto_increment,
	PRODUCTNAME VARCHAR2(255),
	PRODUCTCATEGORY VARCHAR2(255),
	PRODUCTDESCRIPTION VARCHAR2(255),
	PRODUCTPRICE DOUBLE,
	PRODUCTCONDITION VARCHAR2(255),
	PRODUCTSTATUS VARCHAR2(255),
	UNITINSTOCK INT,
	PRODUCTMANUFACTURER VARCHAR2(255)
);
